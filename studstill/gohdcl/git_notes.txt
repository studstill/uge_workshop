git init = create a new repository
git add <filename> = adds the file to the repository staging area
git commit -m "Commit Message" = commit
git commit -a -m "Commit Message" = commit changes from all working files
git status = check the repository status for changes
git log = check the history of the repository
git checkout <commitID> = go to that point in the repository history
git branch <branchName> = create a new branch
git checkout <branchName> = switch to branch
git checkout -b <branchName> = creates a new branch and switches to it
git branch = lists all branches in the repository
git branch -D <branchName> = deletes the branch (cannot be inside it)
git merge <otherBranch> = merges otherBranch into the current branch
git clone <fromRepository> <toRepository> = clone fromRepository and name it toRepository
git remote = show remote repositories
git remote add <remoteRepositoryName> <locationOfRemoteRepository> = add a remote repository to the current repository
git push = pushes changes from your repository to a remote repository; origin is default remote
git push <remoteRepositoryName> <branchName> = pushes to the remote repository; branchName is only required if the branch does not exist on the remote repository
